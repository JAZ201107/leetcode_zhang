My Solution

```java
class Solution {
    public List<List<Integer>> threeSum(int[] nums) {
        Arrays.sort(nums); // sort the array
        
        List<List<Integer>> list = new ArrayList<>();
        
        for(int i = 0; i < nums.length - 2; i++){
            
            if(i>0
              && nums[i] == nums[i-1]){
                continue; 
            }// skip the duplicate
            
            for(int b = i+1, e = nums.length - 1; b < e;){
                if(nums[i] + nums[b] + nums[e] == 0){
                    list.add(Arrays.asList(nums[i], nums[e], nums[b]));
                    b++;
                    e--; 
                    
                    // skip duolicate
                    while(b < e
                         && nums[b]==nums[b-1]){
                        b++;
                    }
                    while(b < e
                         && nums[e] == nums[e+1]){
                        e--;
                    }
                }else if(nums[i] + nums[e] + nums[b] > 0){
                    e--;
                }else {
                    b++;
                }
            }
        }
        
        
        return list;
    }
}
```
